14 !! Sorts XDONT into increasing order (Insertion sort)
15 !!__________________________________________________________
16 !! This subroutine uses insertion sort. It does not use any
17 !! work array and is faster when XDONT is of very small size
18 !! (< 20), or already almost sorted, but worst case behavior
19 !! can happen fairly probably (initially inverse sorted).
20 !! In many cases, the quicksort or merge sort method is faster.
21 !! Michel Olagnon - Apr. 2000
22 !!__________________________________________________________
23 !!__________________________________________________________
24 ! __________________________________________________________
25  Real (kind=kdp), Dimension (:), Intent (InOut) :: xdont
26 ! __________________________________________________________
27  Real (Kind=kdp) :: xwrk, xmin
28 !
29 ! __________________________________________________________
30 !
31  Integer :: ICRS, IDCR, NDON
32 !
33  ndon = Size (xdont)
34 !
35 ! We first bring the minimum to the first location in the array.
36 ! That way, we will have a "guard", and when looking for the
37 ! right place to insert a value, no loop test is necessary.
38 !
39  If (xdont(1) < xdont(ndon)) Then
40  xmin = xdont(1)
41  Else
42  xmin = xdont(ndon)
43  xdont(ndon) = xdont(1)
44  Endif
45  Do idcr = ndon-1, 2, -1
46  xwrk = xdont(idcr)
47  IF (xwrk < xmin) Then
48  xdont(idcr) = xmin
49  xmin = xwrk
50  End If
51  End Do
52  xdont(1) = xmin
53 !
54 ! The first value is now the minimum
55 ! Loop over the array, and when a value is smaller than
56 ! the previous one, loop down to insert it at its right place.
57 !
58  Do icrs = 3, ndon
59  xwrk = xdont(icrs)
60  idcr = icrs - 1
61  If (xwrk < xdont(idcr)) Then
62  xdont(icrs) = xdont(idcr)
63  idcr = idcr - 1
64  Do
65  If (xwrk >= xdont(idcr)) Exit
66  xdont(idcr+1) = xdont(idcr)
67  idcr = idcr - 1
68  End Do
69  xdont(idcr+1) = xwrk
70  End If
71  End Do
72 !
73  Return

15 !! Permute array XVALT randomly, but leaving elements close
16 !! to their initial locations (nearbyness is controled by PCLS).
17 !!_________________________________________________________________
18 !! The routine takes the 1...size(XVALT) index array as real
19 !! values, takes a combination of these values and of random
20 !! values as a perturbation of the index array, and sorts the
21 !! initial set according to the ranks of these perturbated indices.
22 !! The relative proportion of initial order and random order
23 !! is 1-PCLS / PCLS, thus when PCLS = 0, there is no change in
24 !! the order whereas the new order is fully random when PCLS = 1.
25 !! Michel Olagnon - May 2000.
26 !!_________________________________________________________________
27 !!__________________________________________________________
28  Real (kind=kdp), Dimension (:), Intent (InOut) :: xdont
29  Real, Intent (In) :: PCLS
30 ! __________________________________________________________
31 !
32  Real, Dimension (Size(XDONT)) :: XINDT
33  Integer, Dimension (Size(XDONT)) :: JWRKT
34  Real :: PWRK
35  Integer :: I
36 !
37  Call random_number (xindt(:))
38  pwrk = min(max(0.0, pcls), 1.0)
39  xindt = real(Size(xdont)) * xindt
40  xindt = pwrk*xindt + (1.0-pwrk)*(/ (real(i), i=1,size(xdont)) /)
41  Call mrgrnk (xindt, jwrkt)
42  xdont = xdont(jwrkt)

14 !! Sorts partially XDONT, bringing the NORD lowest values at the
15 !! begining of the array
16 !!__________________________________________________________
17 !! This subroutine uses insertion sort, limiting insertion
18 !! to the first NORD values. It does not use any work array
19 !! and is faster when NORD is very small (2-5), but worst case
20 !! behavior can happen fairly probably (initially inverse sorted)
21 !! In many cases, the refined quicksort method is faster.
22 !! Michel Olagnon - Feb. 2000
23 !!__________________________________________________________
24 !!__________________________________________________________
25  Real (kind=kdp), Dimension (:), Intent (InOut) :: xdont
26  Integer, Intent (In) :: NORD
27 ! __________________________________________________________
28  Real (kind=kdp) :: xwrk, xwrk1
29 !
30  Integer :: ICRS, IDCR
31 !
32  Do icrs = 2, nord
33  xwrk = xdont(icrs)
34  Do idcr = icrs - 1, 1, - 1
35  If (xwrk >= xdont(idcr)) Exit
36  xdont(idcr+1) = xdont(idcr)
37  End Do
38  xdont(idcr+1) = xwrk
39  End Do
40 !
41  xwrk1 = xdont(nord)
42  Do icrs = nord + 1, SIZE (xdont)
43  If (xdont(icrs) < xwrk1) Then
44  xwrk = xdont(icrs)
45  xdont(icrs) = xwrk1

53  End Do
